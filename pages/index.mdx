import { getPagesUnderRoute } from 'nextra/context'
import Link from 'next/link'

export const getStaticProps = () => {
  return {
    props: {
      posts: getPagesUnderRoute('/posts')
        .map((page) => ({
          ...page,
          date: page.frontMatter?.date || page.meta?.date
        }))
        .filter(page => page.date && page.route !== '/posts')
        .sort((a, b) => new Date(b.date) - new Date(a.date))
        .slice(0, 5)
    }
  }
}
